\begin{python}
import torch
from mylib import Environment
from mylib import PolicyNet

# Set up environment and model.
env = Environment(
    ndim=4, height=8
)
model = PolicyNet(
    input_dim=env.obs_dim,
    hidden_dim=256,
    output_dim=env.n_actions,
)
opt = torch.optim.Adam(
    model.parameters(), lr=1e-3
)

# Training loop.
for epoch in range(1000):
    traj = env.sample(model, n=16)
    loss = model.loss(traj)
    opt.zero_grad()
    loss.backward()
    opt.step()
\end{python}

\vspace{1cm}
This enables users to swap single elements of their training pipeline without
changes to the rest of their code.
